# --- PASSO 1: IMPORTANDO AS FERRAMENTAS ---
import streamlit as st
import pandas as pd
import plotly.express as px 

# --- PASSO 2: CONFIGURANDO A PÁGINA ---
# Esta linha define o título na aba do navegador, o ícone e deixa a tela larga
st.set_page_config(page_title="Brasileirão_dados", page_icon="⚽", layout="wide")   

# --- PASSO 3: CARREGANDO E LIMPANDO OS DADOS ---
# Guarda os dados na memória para o painel não ficar lento
@st.cache_data
def carregar_dados():
    tabela = pd.read_excel("Brasileirao_2006_2025.xlsx")    
    return tabela

df = carregar_dados()   

# Limpa espaços invisíveis nos nomes dos times (ex: transforma "Flamengo " em "Flamengo")
df["Time"] = df["Time"].str.strip() 


# --- PASSO 4: CABEÇALHO (IMAGEM E TÍTULO) ---
# Divide o topo em duas colunas (tamanho 1 para a taça, tamanho 8 para o texto)
col1, col2 = st.columns([1, 8])

with col1:
    # Mostra a imagem
    st.image("taca.png", width=70)  

with col2:
    # Mostra os títulos com a cor azul
    st.title((":blue[Brasileirão de 2006 a 2025]"))
    st.subheader(":blue[Série A do Campeonato Brasileiro]")


# --- PASSO 5: MENU LATERAL E FILTROS ---
st.sidebar.header("Filtros")

# Puxa os anos únicos da base e organiza do maior para o menor
lista_anos = df["Ano"].unique()
lista_anos_ordenada = sorted(lista_anos, reverse=True)

# Cria a caixinha de seleção na esquerda
ano_selecionado = st.sidebar.selectbox(" Selecione a temporada ", lista_anos_ordenada)  

# Corta a base de dados para deixar só os dados do ano escolhido
df_filtrado = df[df["Ano"] == ano_selecionado]


# --- PASSO 6: INTELIGÊNCIA DOS CARTÕES (KPIs) ---

# Descobrindo o Campeão (Ordena por pontos e pega a primeira posição [0])
df_ordenado = df_filtrado.sort_values(by="Pontos", ascending=False)
nome_campeao = df_ordenado.iloc[0]["Time"]
pontos_campeao = df_ordenado.iloc[0]["Pontos"]

# Descobrindo o Melhor Ataque (Acha o máximo de gols pró)
indice_melhor_ataque = df_filtrado["Gols_Pro"].idxmax()
time_melhor_ataque = df_filtrado.loc[indice_melhor_ataque, "Time"]
gols_melhor_ataque = df_filtrado.loc[indice_melhor_ataque, "Gols_Pro"]

# Descobrindo a Melhor Defesa (Acha o mínimo de gols sofridos)
indice_melhor_defesa = df_filtrado["Gols_Sofridos"].idxmin()
time_melhor_defesa = df_filtrado.loc[indice_melhor_defesa, "Time"]
gols_melhor_defesa = df_filtrado.loc[indice_melhor_defesa, "Gols_Sofridos"]

# Desenhando os cartões na tela
st.write("---") 

# Linha 1 de cartões: divide a tela em 3 e coloca as métricas
kpi1, kpi2, kpi3 = st.columns(3)
with kpi1:
    st.metric(label="🏆 Campeão", value=nome_campeao)
with kpi2:
    st.metric(label="⭐ Pontos do Campeão", value=f"{pontos_campeao} pts")
with kpi3:
    st.metric(label="📅 Temporada", value=ano_selecionado)

# Linha 2 de cartões: divide a tela em 2 e coloca as métricas
kpi4, kpi5 = st.columns(2)
with kpi4:
    st.metric(label="⚽ Melhor Ataque", value=time_melhor_ataque, delta=f"{gols_melhor_ataque} gols marcados", delta_color="normal")
with kpi5:
    st.metric(label="🛡️ Melhor Defesa", value=time_melhor_defesa, delta=f"{gols_melhor_defesa} gols sofridos", delta_color="inverse")

st.write("---") 


# --- PASSO 7: TABELA DE CLASSIFICAÇÃO ---
st.subheader(f"Tabela de classificação {ano_selecionado}")  

# Plota a tabela na tela escondendo o índice padrão do Python
st.dataframe(df_filtrado[["Posicao", "Time", "Pontos", "Jogos", "Vitorias", "Empates", "Derrotas"]], hide_index=True, use_container_width=True) 

st.write("----")    


# --- PASSO 8: GRÁFICO DE BARRAS COLORIDO ---
st.subheader("📊 Comparativo de Pontos")    

# Dicionário com os códigos hexadecimais das cores de cada time
cores_times = {
    "Flamengo": "#c22a1e",      
    "Palmeiras": "#006437",     
    "São Paulo": "#FE0000",     
    "Corinthians": "#000000",   
    "Cruzeiro": "#003a70",      
    "Atlético-MG": "#000000",   
    "Grêmio": "#0d80bf",        
    "Internacional": "#e50000", 
    "Fluminense": "#7a0016",    
    "Botafogo": "#000000",      
    "Vasco": "#000000",         
    "Santos": "#ffffff",        
    "Bahia": "#003a70",         
    "Fortaleza": "#1d3989",     
    "Athletico-PR": "#c8102e",  
    "Bragantino": "#000000",    
    "Cuiabá": "#fdf228",        
    "Juventude": "#006437",     
    "Vitória": "#cc0000",       
    "Mirassol": "#fdf228"       
}

# Configura o gráfico de barras
fig = px.bar(
    df_filtrado,
    x="Time",
    y="Pontos",
    color="Time", 
    title=f"Pontuação dos Times - {ano_selecionado}",
    text_auto = True, 
    color_discrete_map=cores_times # Pinta cada barra com a cor certa do dicionário acima
)       

# Desenha uma bordinha preta nas barras
fig.update_traces(marker_line_color="black", marker_line_width=1)

# Imprime o gráfico na tela
st.plotly_chart(fig, use_container_width=True)
